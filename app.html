async function loadExternalFileFromQuery() { const params = new URLSearchParams(location.search); const fileName = params.get('file'); if (!fileName) return false; const res = await fetch(`./${encodeURIComponent(fileName)}?v=${Date.now()}`, { cache: 'no-store' }); if (!res.ok) throw new Error(`파일을 읽지 못했습니다: ${fileName}`); const buffer = await res.arrayBuffer(); const rows = await parseFetchedFile(buffer, fileName); const parsed = extractSchedule(rows, fileName); app.records = parsed.records; app.names = parsed.names; app.year = parsed.year; app.month = parsed.month; bindSelectors(); applyAutoSelectionFromFileName(fileName); renderSummary(parsed); render(); return true; }
